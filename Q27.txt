SELECT Track.TrackId, MediaType.Name FROM InvoiceLine JOIN Track ON Track.TrackId == InvoiceLine.TrackId JOIN Album ON Track.AlbumId == Album.AlbumId JOIN Artist ON Album.ArtistId == Artist.ArtistId JOIN MediaType ON Track.MediaTypeId == MediaType.MediaTypeId JOIN Genre ON Track.GenreId == Genre.GenreId GROUP BY MediaType.Name ORDER BY Track.TrackId DESC LIMIT(1)
